Example of *ngSwitch vs switch
@switch (color) { @case ('red') {

The color is red.

} @case ('blue') {

The color is blue.

} @default {

Unknown color

} }
@switch (userRole) { @case ('admin') {

Welcome, Admin!

} @case ('editor') {

Welcome, Editor!

} @case ('viewer') {

Welcome, Viewer!

} @default {

Unknown role

} }
